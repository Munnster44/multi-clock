<script>
function updateClocks() {
    const zones = {
        pt: "America/Vancouver",
        mt: "America/Edmonton",
        ct: "America/Winnipeg",
        et: "America/Toronto",
        at: "America/Halifax",
        nt: "America/St_Johns"
    };

    for (const id in zones) {
        const time = new Date().toLocaleTimeString("en-CA", {
            timeZone: zones[id],
            hour: "2-digit",
            minute: "2-digit",
            second: "2-digit"
        });
        document.getElementById(id).textContent = time;
    }
}

setInterval(updateClocks, 1000);
updateClocks();
</script>
